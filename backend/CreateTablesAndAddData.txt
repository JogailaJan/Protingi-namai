CREATE TABLE Systems (
    id INTEGER PRIMARY KEY,
    name TEXT NOT NULL,
    link TEXT NOT NULL,
    description TEXT
    -- Add other relevant columns as needed
);

CREATE TABLE ServiceAreas (
    id INTEGER PRIMARY KEY,
    name TEXT NOT NULL,
    value TEXT NOT NULL,
    description TEXT
    -- Add other relevant columns as needed
);

CREATE TABLE Functionalities (
    id INTEGER PRIMARY KEY,
    name TEXT NOT NULL,
    group_name TEXT NOT NULL,
    value TEXT NOT NULL,
    description TEXT
    -- Add other relevant columns as needed
);

CREATE TABLE SystemServiceAreas (
    system_id INTEGER,
    service_area_id INTEGER,
    PRIMARY KEY (system_id, service_area_id),
    FOREIGN KEY (system_id) REFERENCES Systems(id),
    FOREIGN KEY (service_area_id) REFERENCES ServiceAreas(id)
);

CREATE TABLE ServiceAreaFunctionalities (
    service_area_id INTEGER,
    functionality_id INTEGER,
    PRIMARY KEY (service_area_id, functionality_id),
    FOREIGN KEY (service_area_id) REFERENCES ServiceAreas(id),
    FOREIGN KEY (functionality_id) REFERENCES Functionalities(id)
);

CREATE TABLE CommunicationStandards (
    id INTEGER PRIMARY KEY,
    system_id INTEGER,
    standard TEXT NOT NULL,
    FOREIGN KEY (system_id) REFERENCES Systems(id)
);
